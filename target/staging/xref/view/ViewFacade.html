251         /**
252          * <h3>The Class MoveAction provides the movement of an action.</h3>
253          * @author chris
254          *
255          */
256         public class MoveAction extends AbstractAction {
257         	 /** the serialVersionUID. */
258 			private static final long serialVersionUID = 1L;
259 			 /** the xDelta and yDelta are the coordinates in a map. */
260 			private final int xDelta, yDelta;
261 			 /** the level of the map */
262 			private final String level;
263 			
264 			/**
265 			 * the constructor MoveAction			 
266 			 * @param xDelta
267 			 * @param yDelta
268 			 * @param level
269 			 */
270             public MoveAction(int xDelta, int yDelta, String level) {
271                 this.xDelta = xDelta;
272                 this.yDelta = yDelta;
273                 this.level = level;
274             }
275 
276             @Override
277             public void actionPerformed(ActionEvent e) {
278             	
279                 int index = getComponentZOrder(l), index2 = getComponentZOrder(l);
280                 index += xDelta;
281                 index += (yDelta * 20);
282                 if (index < 0) {
283                     index = 0;
284                 } else if (index >= getComponentCount()) {
285                     index = getComponentCount() - 1;
286                 }
287                 if (map[(index-(index%20))/20][index%20] == 'N') {
288 	                setComponentZOrder(getComponent(index), index2);
289 	                map[(index2 -(index2%20))/20][index2%20] = 'N';
290 	                setComponentZOrder(l, index);
291 	                map[(index-(index%20))/20][index%20] = 'A';
292                 }
293                 else if (map[(index-(index%20))/20][index%20] == 'C') {
294                 	JLabel empt = new JLabel();
295                     empt.setIcon(new ImageIcon(new ImageIcon("sprite/empty.png").getImage().getScaledInstance(64, 64, Image.SCALE_DEFAULT)));
296 	                remove(index);
297 	                add(empt, index2);
298 	                JLabel go = new JLabel();
299                     go.setIcon(new ImageIcon(new ImageIcon("sprite/gate_open.png").getImage().getScaledInstance(64, 64, Image.SCALE_DEFAULT)));
300 	                for (int y = 0; y < 12; y++) {
301 	                	for (int x = 0; x < 20; x++) {
302 	                		if (level.charAt(x+y*20) == 'D') {
303 	                			remove(x+y*20);
304 	                			add(go, (x+y*20));
305 	                		}
306 	                	}
307 	                }
308 	                map[(index2 -(index2%20))/20][index2%20] = 'N';
309 	                setComponentZOrder(l, index);
310 	                map[(index-(index%20))/20][index%20] = 'A';
311                 }
312                 else if (map[(index-(index%20))/20][index%20] == 'M') {
313                 	points += 65;
314                 	JLabel t = new JLabel(Integer.toString(points));
315             		Font font = new Font("Arial",Font.BOLD,30);
316             		t.setForeground(Color.red);
317             		t.setFont(font);
318                 	remove(250);
319                 	add(t, 250);
320                 	JLabel empt = new JLabel();
321                     empt.setIcon(new ImageIcon(new ImageIcon("sprite/empty.png").getImage().getScaledInstance(64, 64, Image.SCALE_DEFAULT)));
322 	                remove(index);
323 	                add(empt, index2);
324 	                map[(index2 -(index2%20))/20][index2%20] = 'N';
325 	                setComponentZOrder(l, index);
326 	                map[(index-(index%20))/20][index%20] = 'A';
327                 }
328                 
329                 revalidate();
330                 repaint();
331             }
332 
333         }
